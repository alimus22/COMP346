/**

 * try {

 *     // or from some other IO activity like temporary files, etc.

 *     if (stream.isCauseOf(e)) {

 * }
 * </pre>

 *     processStream(stream);
 * } catch (IOException e) {
 *     stream.throwIfCauseOf(e);

 * }
